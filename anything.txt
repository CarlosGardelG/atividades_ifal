Questão 04: Escreva um programa com funções para calcular a área (b x h) e o perímetro de um retângulo (2b+2h). Forneça as seguintes opções ao usuário:
1. Calcular a área do retângulo
2. Calcular o perímetro do retângulo*/
import 'dart:io';   
void calculateRectangle(double base, double height, void Function(double, double) callback) {
  double area = base * height;
  double perimeter = 2 * base + 2 * height;
  callback(area, perimeter);
}

void main() {
  print('Digite a base do retângulo:');
  double base = double.parse(stdin.readLineSync()!);
  
  print('Digite a altura do retângulo:');
  double height = double.parse(stdin.readLineSync()!);
  
  calculateRectangle(base, height, (area, perimeter) {
    print('Área: $area');
    print('Perímetro: $perimeter');
  });


Map<String, double> calcularRetangulo(double base, double altura) {
  double area = base * altura;
  double perimetro = 2 * (base + altura);

  return {
    'area': area,
    'perimetro': perimetro,
  };
}

void main() {
  double base = 5;
  double altura = 3;

  var resultado = calcularRetangulo(base, altura);

  double area = resultado['area']!;
  double perimetro = resultado['perimetro']!;

  print('Área: $area');
  print('Perímetro: $perimetro');
}

(double, double) calcularRetangulo(double base, double altura) {
  double area = base * altura;
  double perimetro = 2 * (base + altura);

  return (area, perimetro);
}

void main() {
  double base = 5;
  double altura = 3;

  var (area, perimetro) = calcularRetangulo(base, altura);

  print('Área: $area');
  print('Perímetro: $perimetro');
}
